#include <stdio.h>
#include <stdlib.h>

void strrev(char* str) {
    char* s = str;
    char* e = str + strlen(str) - 1;
    while(s < e) {
        char temp = *s;
        *s = *e ;
        *e = temp;
        s++;
        e--;
    }
}

void my_strcpy(char *dst, const char* src) {
    char* org = dst;
    
    while(*src != '\0') {
        *dst++ = *src++;
    }
    *dst = '\0';
    return org;
}

void my_strcat(char* s1, char*s2) {
    char* d = s1;
    while(*d != '\0') {
        d++;   
    }
    
    while(*s2) {
        *d++ = *s2++;
    }
    *d = '\0';
    return;
}

int reverseNumber(int num) {
    int rev =0;
    while(num > 0) {
        rev = (rev*10) + (num % 10);
        num = num / 10;
    }
    return rev;
}

int factorial(int num) {
    if(num < 0) return -1;
    if(num <= 1) return 1;
    return num * factorial(num - 1);
}

int multiply_wo_asterik(int a, int b) {
    int res = 0;
    
    for(int i = 0; i < b; i++) {
        res = res + a;
    }
    return res;
}
